class C{
    public static void main(String args[]) {
        if (args.length < 4) {
            System.out.println("Usage: java BMI <FirstName> <LastName> <Weight(kg)> <Height(m)>");
            return;
        }

        // Reading values from command line arguments
        String firstName = args[0];
        String lastName = args[1];
        double weight = Double.parseDouble(args[2]);  // in kilograms
        double height = Double.parseDouble(args[3]);  // in meters

        // BMI calculation
        double bmi = weight / (height * height);

        // Display result
        System.out.println("Name: " + firstName + " " + lastName);
        System.out.println("Weight: " + weight + " kg");
        System.out.println("Height: " + height + " m");
        System.out.println("BMI Index: " + bmi);
    }
}
