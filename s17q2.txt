import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

 class StringOperations extends JFrame implements ActionListener {

    JTextField t1, t2, t3;
    JButton b1, b2;

    StringOperations() {
        // Frame settings
        setTitle("String Operations");
        setSize(400, 200);
        setLayout(new GridLayout(4, 2, 5, 5));
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        // Create labels and text fields
        add(new JLabel("String 1:"));
        t1 = new JTextField();
        add(t1);

        add(new JLabel("String 2:"));
        t2 = new JTextField();
        add(t2);

        add(new JLabel("Result:"));
        t3 = new JTextField();
        t3.setEditable(false);  // result textbox should not be editable
        add(t3);

        // Buttons
        b1 = new JButton("Concatenate");
        b1.addActionListener(this);
        add(b1);

        b2 = new JButton("Reverse String 1");
        b2.addActionListener(this);
        add(b2);

        setVisible(true);
    }

    public void actionPerformed(ActionEvent e) {
        String str1 = t1.getText();
        String str2 = t2.getText();

        if (e.getSource() == b1) {
            // Concatenate strings
            t3.setText(str1 + str2);
        } else if (e.getSource() == b2) {
            // Reverse first string
            String reversed = new StringBuilder(str1).reverse().toString();
            t3.setText(reversed);
        }
    }

    public static void main(String[] args) {
        new StringOperations();
    }
}
